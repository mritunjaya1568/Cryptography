# 60 =

```c
#include<stdio.h>
#include<unistd.h>
#include<sys/wait.h>
#include<fcntl.h>


int pwncollege()
{
	int fd[2];
	if (pipe(fd) == -1)
	{
		return 2;
	}

	int pid1 = fork();
	if (pid1 < 0)
	{
		return 1;
	}
	if (pid1 == 0)
	{
		// child process 1 - this will write to cat
		close(fd[0]);
		dup2(fd[1], STDOUT_FILENO);
		char *argv[] = {
			"/challenge/embryoio_level60",NULL
		};
		close(fd[1]);
		execv("/challenge/embryoio_level60",argv);
	}

	int pid2 = fork();
	if (pid2 < 0)
	{
		return 3;
	}
	if(pid2 == 0)
	{
		// child process 2 - cat - this will read its input from child process 1
		close(fd[1]);
		dup2(fd[0],STDIN_FILENO);
		char *argv[] = {
			"cat",NULL
		};
		close(fd[0]);
		execvp("cat",argv);
	}

	close(fd[0]);
	close(fd[1]);

	waitpid(pid1,NULL,0);
	waitpid(pid2,NULL,0);
}

void main(int argc , char * argv[])
{
	pwncollege();
}
```
